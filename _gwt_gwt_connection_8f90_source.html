478  subroutine gwtgwtcon_fc(this, kiter, iasln, amatsln, rhssln, inwtflag)
479  class(gwtgwtconnectiontype) :: this !< the connection
480  integer(I4B), intent(in) :: kiter !< the iteration counter
481  integer(I4B), dimension(:), intent(in) :: iasln !< global system's IA array
482  real(DP), dimension(:), intent(inout) :: amatsln !< global system matrix coefficients
483  real(DP), dimension(:), intent(inout) ::rhssln !< global right-hand-side
484  integer(I4B), optional, intent(in) :: inwtflag !< newton-raphson flag
485  ! local
486  integer(I4B) :: n, nglo, ipos
487 
488  call this%gwtInterfaceModel%model_fc(kiter, this%amat, this%nja, inwtflag)
489 
490  ! map back to solution matrix
491  do n = 1, this%neq
492  ! We only need the coefficients for our own model
493  ! (i.e. rows in the matrix that belong to this%owner):
494  if (.not. associated(this%gridConnection%idxToGlobal(n)%model, &
495  this%owner)) then
496  cycle
497  end if
498 
499  nglo = this%gridConnection%idxToGlobal(n)%index + &
500  this%gridConnection%idxToGlobal(n)%model%moffset
501  rhssln(nglo) = rhssln(nglo) + this%rhs(n)
502 
503  do ipos = this%ia(n), this%ia(n + 1) - 1
504  amatsln(this%mapIdxToSln(ipos)) = amatsln(this%mapIdxToSln(ipos)) + &
505  this%amat(ipos)
506  end do
507  end do
508 
509  ! FC the movers through the exchange; we can call
510  ! exg_fc() directly because it only handles mover terms (unlike in GwfExchange%exg_fc)
511  if (this%exchangeIsOwned) then
512  call this%gwtExchange%exg_fc(kiter, iasln, amatsln, rhssln, inwtflag)
513  end if
514 
515  end subroutine gwtgwtcon_fc
