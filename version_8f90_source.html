75  !!
76  !! Write header for program to the program listing file.
77  !!
78  !<
79  subroutine write_listfile_header(iout, cmodel_type, write_sys_command, &
80  write_kind_info)
81  ! -- dummy variables
82  integer(I4B), intent(in) :: iout !< program listing file
83  character(len=*), intent(in), optional :: cmodel_type !< optional model type string
84  logical(LGP), intent(in), optional :: write_sys_command !< boolean indicating if the system command should be written
85  logical(LGP), intent(in), optional :: write_kind_info !< boolean indicating in program data types should be written
86  ! -- local variables
87  character(len=LENBIGLINE) :: syscmd
88  character(len=LENBIGLINE) :: compiler
89  character(len=LENBIGLINE) :: compiler_options
90  integer(I4B) :: iheader_width = 80
91  logical(LGP) :: wki
92  logical(LGP) :: wsc
93  !
94  ! -- Write title to list file
95  call write_centered('MODFLOW'//mfvnam, iheader_width, iunit=iout)
96  call write_centered(mftitle, iheader_width, iunit=iout)
97  !
98  ! -- Write model type to list file
99  if (present(cmodel_type)) then
100  call write_centered(cmodel_type, iheader_width, iunit=iout)
101  end if
102  !
103  ! -- Write version
104  call write_centered('VERSION '//version, iheader_width, iunit=iout)
105  !
106  ! -- Write if develop mode
107  if (idevelopmode == 1) then
108  call write_centered('***DEVELOP MODE***', iheader_width, iunit=iout)
109  end if
110  !
111  ! -- Write compiler version
112  call get_compiler(compiler)
113  call write_centered(' ', iheader_width, iunit=iout)
114  call write_centered(trim(adjustl(compiler)), iheader_width, iunit=iout)
115  !
116  ! -- Write disclaimer
117  write (iout, fmtdisclaimer)
118  !
119  ! -- Write license information
120  if (iout /= istdout) then
121  write (iout, fmtlicense)
122  end if
123  !
124  ! -- write compiler options
125  if (iout /= istdout) then
126  call get_compile_options(compiler_options)
127  call write_message(compiler_options, iunit=iout)
128  end if
129  !
130  ! -- Write the system command used to initiate simulation
131  wsc = .true.
132  if (present(write_sys_command)) wsc = write_sys_command
133  if (wsc) then
134  call get_command(syscmd)
135  write (iout, '(/,a,/,a)') &
136  'System command used to initiate simulation:', trim(syscmd)
137  end if
138  !
139  ! -- Write precision of real variables
140  wki = .true.
141  if (present(write_kind_info)) wki = write_kind_info
142  if (wki) then
143  write (iout, '(/,a)') 'MODFLOW was compiled using uniform precision.'
144  call write_kindinfo(iout)
145  end if
146  write (iout, *)
147  !
148  ! -- return
149  return
150  end subroutine write_listfile_header
