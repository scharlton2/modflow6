158  !> @brief Calculate the wetted perimeters for each line segment
159  !!
160  !! Subroutine to calculate the wetted perimeter for each line segment
161  !! that defines the reach using the cross-section station elevation
162  !! data given a passed elevation.
163  !!
164  !<
165  subroutine get_wetted_perimeters(npts, x, z, d, p)
166  ! -- dummy variables
167  integer(I4B), intent(in) :: npts !< number of station elevation data for a reach
168  real(DP), dimension(npts), intent(in) :: x !< cross-section station distances (x-distance)
169  real(DP), dimension(npts), intent(in) :: z !< cross-section elevation data
170  real(DP), intent(in) :: d !< depth to evaluate cross-section
171  real(DP), dimension(npts-1) :: p !< wetted perimeter for each line segment
172  ! -- local variables
173  integer(I4B) :: n
174  real(DP) :: x0
175  real(DP) :: x1
176  real(DP) :: z0
177  real(DP) :: z1
178  real(DP) :: zmax
179  real(DP) :: zmin
180  real(DP) :: e
181  real(DP) :: xlen
182  real(DP) :: zlen
183  !
184  ! -- iterate over the station-elevation data
185  do n = 1, npts - 1
186  !
187  ! -- initialize the wetted perimeter
188  p(n) = dzero
189  !
190  ! -- initialize station-elevation data for segment
191  x0 = x(n)
192  x1 = x(n+1)
193  z0 = z(n)
194  z1 = z(n+1)
195  !
196  ! -- get the start and end station position of the wetted segment
197  call get_wetted_station(x0, x1, z0, z1, zmax, zmin, d)
198  !
199  ! -- calculate the elevation to evaluate
200  e = zmin + d
201  !
202  ! -- calculate the wetted perimeter for the segment
203  xlen = x1 - x0
204  if (xlen > dzero) then
205  if (e > zmax) then
206  zlen = zmax - zmin
207  else
208  zlen = e - zmin
209  end if
210  p(n) = sqrt(xlen**dtwo + zlen**dtwo)
211  end if
212  end do
213  !
214  ! -- return
215  return
216  end subroutine get_wetted_perimeters

218  !> @brief Calculate the cross-sectional areas for each line segment
219  !!
220  !! Subroutine to calculate the cross-sectional area for each line segment
221  !! that defines the reach using the cross-section station elevation
222  !! data given a passed elevation.
223  !!
224  !<
225  subroutine get_cross_section_areas(npts, x, z, d, a)
226  ! -- dummy variables
227  integer(I4B), intent(in) :: npts !< number of station elevation data for a reach
228  real(DP), dimension(npts), intent(in) :: x !< cross-section station distances (x-distance)
229  real(DP), dimension(npts), intent(in) :: z !< cross-section elevation data
230  real(DP), intent(in) :: d !< depth to evaluate cross-section
231  real(DP), dimension(npts-1) :: a !< cross-sectional area for each line segment
232  ! -- local variables
233  integer(I4B) :: n
234  real(DP) :: x0
235  real(DP) :: x1
236  real(DP) :: z0
237  real(DP) :: z1
238  real(DP) :: zmax
239  real(DP) :: zmin
240  real(DP) :: e
241  real(DP) :: xlen
242  !
243  ! -- iterate over the station-elevation data
244  do n = 1, npts - 1
245  !
246  ! -- initialize the cross-sectional area
247  a(n) = dzero
248  !
249  ! -- initialize station-elevation data for segment
250  x0 = x(n)
251  x1 = x(n+1)
252  z0 = z(n)
253  z1 = z(n+1)
254  !
255  ! -- get the start and end station position of the wetted segment
256  call get_wetted_station(x0, x1, z0, z1, zmax, zmin, d)
257  !
258  ! -- calculate the elevation to evaluate
259  e = zmin + d
260  !
261  ! -- calculate the cross-sectional area for the segment
262  xlen = x1 - x0
263  if (xlen > dzero) then
264  !
265  ! -- add the area above zmax
266  if (e > zmax) then
267  a(n) = xlen * (e - zmax)
268  end if
269  !
270  ! -- add the area below zmax
271  if (zmax /= zmin .and. e > zmin) then
272  a(n) = a(n) + dhalf * (e - zmin)
273  end if
274  end if
275  end do
276  !
277  ! -- return
278  return
279  end subroutine get_cross_section_areas

325  !> @brief Calculate the station values for the wetted portion of the cross-section
326  !!
327  !! Subroutine to calculate the station values that define the extent of the
328  !! wetted portion of the cross section for a line segment. The left (x0) and
329  !! right (x1) station positions are altered if the passed elevation is less
330  !! than the maximum line segment elevation. If the line segment is dry the left
331  !! and right station are equal. Otherwise the wetted station values are equal
332  !! to the full line segment or smaller if the passed elevation is less than
333  !! the maximum line segment elevation.
334  !!
335  !<
336  pure subroutine get_wetted_station(x0, x1, z0, z1, zmax, zmin, d)
337  ! -- dummy variables
338  real(dp), intent(inout) :: x0 !< left station position
339  real(dp), intent(inout) :: x1 !< right station position
340  real(dp), intent(in) :: z0 !< elevation at the left station
341  real(dp), intent(in) :: z1 !< elevation at the right station
342  real(dp), intent(inout) :: zmax !< maximum elevation
343  real(dp), intent(inout) :: zmin !< minimum elevation
344  real(dp), intent(in) :: d !< depth to evaluate cross-section
345  ! -- local variables
346  real(dp) :: e
347  real(dp) :: xlen
348  real(dp) :: zlen
349  real(dp) :: slope
350  real(dp) :: dx
351  real(dp) :: xt
352  real(dp) :: xt0
353  real(dp) :: xt1
354  !
355  ! -- calculate the minimum and maximum elevation
356  zmin = min(z0, z1)
357  zmax = max(z0, z1)
358  !
359  ! -- calculate the elevation to evaluate
360  e = zmin + d
361  !
362  ! -- if e is less than or equal to the minimum value the
363  ! station length (xlen) is zero
364  if (e <= zmin) then
365  x1 = x0
366  ! -- if e is between zmin and zmax station length is less
367  ! than z1 - z0
368  else if (e < zmax) then
369  xlen = x1 - x0
370  zlen = z1 - z0
371  if (abs(zlen) > 0.) then
372  slope = xlen / zlen
373  else
374  slope = dzero
375  end if
376  if (z0 > z1) then
377  dx = (e - z1) * slope
378  xt = x1 + dx
379  xt0 = xt
380  xt1 = x1
381  else
382  dx = (e - z0) * slope
383  xt = x0 + dx
384  xt0 = x0
385  xt1 = xt
386  end if
387  x0 = xt0
388  x1 = xt1
389  end if
390  !
391  ! -- return
392  return
393  end subroutine get_wetted_station
