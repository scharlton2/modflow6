311 
312  ! for now, connecting the same nodes nrs will be

314  areequal = all(gwfconn%primaryExchange%nodem1 == &

317  gwtconn%primaryExchange%nodem2)
318  if (areequal) then
319  ! same DIS, same exchange: link and go to next GWT conn.
320  write (iout, '(/6a)') 'Linking exchange ', &

323  ' (using interface model) for GWT model ', &
324  trim(gwtmodel%name)

326  call this%link_connections(gwtconn, gwfconn)
327  exit gwfloop
328  end if
329  end if
330  end do gwfloop
331 
332  ! fallback option: coupling to old gwfgwf exchange,
333  ! (this will go obsolete at some point)
334  if (gwfconnidx == -1) then

336  gwfex => getgwfexchangefromlist(baseexchangelist, iex)
337 
338  ! -- There is no guarantee that iex is a gwfExg, in which case

344  ! again, connecting the same nodes nrs will be

354  trim(gwtmodel%name)

357

361  !cdl todo: check and make sure gwtEx has mvt active
362  call gwtconn%gwtExchange%mvt%set_pointer_mvrbudobj( &
363  gwfex%mvr%budobj)
364  end if
365  end if
366

368  gwtconn%gwtInterfaceModel%fmi%flows_from_file = .false.

371  end if
372  end if
373 
374  end do gwfloopexg
375  end if
376

378  ! none found, report

380  ' model ', trim(gwtmodel%name), ' with exchange ', &
381  trim(gwtconn%primaryExchange%name), ' to GWF model ', &
