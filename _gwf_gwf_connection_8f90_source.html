319  !> @brief Write the calculated coefficients into the global
320  !< system matrix and the rhs
321  subroutine gwfgwfcon_fc(this, kiter, iasln, amatsln, rhssln, inwtflag)
322  class(gwfgwfconnectiontype) :: this !< this connection
323  integer(I4B), intent(in) :: kiter !< the iteration counter
324  integer(I4B), dimension(:), intent(in) :: iasln !< global system's IA array
325  real(DP), dimension(:), intent(inout) :: amatsln !< global system matrix coefficients
326  real(DP), dimension(:), intent(inout) ::rhssln !< global right-hand-side
327  integer(I4B), optional, intent(in) :: inwtflag !< newton-raphson flag
328  ! local
329  integer(I4B) :: n, ipos, nglo
330 
331  ! fill (and add to...) coefficients for interface
332  call this%gwfInterfaceModel%model_fc(kiter, this%amat, this%nja, inwtflag)
333 
334  ! map back to solution matrix
335  do n = 1, this%neq
336  ! we cannot check with the mask here, because cross-terms are not
337  ! necessarily from primary connections. But, we only need the coefficients
338  ! for our own model (i.e. fluxes into cells belonging to this%owner):
339  if (.not. associated(this%gridConnection%idxToGlobal(n)%model, &
340  this%owner)) then
341  ! only add connections for own model to global matrix
342  cycle
343  end if
344 
345  nglo = this%gridConnection%idxToGlobal(n)%index + &
346  this%gridConnection%idxToGlobal(n)%model%moffset
347  rhssln(nglo) = rhssln(nglo) + this%rhs(n)
348 
349  do ipos = this%ia(n), this%ia(n + 1) - 1
350  amatsln(this%mapIdxToSln(ipos)) = amatsln(this%mapIdxToSln(ipos)) + &
351  this%amat(ipos)
352  end do
353  end do
354 
355  ! FC the movers through the exchange; we cannot call
356  ! exg_fc() directly because it calculates matrix terms
357  if (this%exchangeIsOwned) then
358  if (this%gwfExchange%inmvr > 0) then
359  call this%gwfExchange%mvr%mvr_fc()
360  end if
361  end if
362 
363  end subroutine gwfgwfcon_fc
