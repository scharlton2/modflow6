139  !> @ brief Copy data read from a budget file into this object
140  !!
141  !! The routine is called when GWF flows are read from a budget file
142  !! created from a previous GWF simulation. Arrays here must be
143  !! dynamically resized because maxbound is not known unless the
144  !! entire budget file was read first.
145  !!
146  !<
147  subroutine copy_values(this, nbound, nodelist, flow, auxvar)
148  class(packagebudgettype) :: this !< PackageBudgetType object
149  integer(I4B), intent(in) :: nbound !< number of entries
150  integer(I4B), dimension(:), contiguous, intent(in) :: nodelist !< array of GWT node numbers
151  real(DP), dimension(:), contiguous, intent(in) :: flow !< array of flow rates
152  real(DP), dimension(:, :), contiguous, intent(in) :: auxvar !< array of auxiliary variables
153  integer(I4B) :: i
154  !
155  ! -- Assign variables
156  this%nbound = nbound
157  !
158  ! -- Lists are not large enough (maxbound is not known), so need to
159  ! reallocate based on size in binary budget file.
160  if (size(this%nodelist) < nbound) then
161  call mem_reallocate(this%nodelist, nbound, 'NODELIST', this%memoryPath)
162  call mem_reallocate(this%flow, nbound, 'FLOW', this%memoryPath)
163  call mem_reallocate(this%auxvar, this%naux, nbound, 'AUXVAR', &
164  this%memoryPath)
165  end if
166  !
167  ! -- Copy values into member variables
168  do i = 1, nbound
169  this%nodelist(i) = nodelist(i)
170  this%flow(i) = flow(i)
171  this%auxvar(:, i) = auxvar(:, i)
172  end do
173  end subroutine copy_values
